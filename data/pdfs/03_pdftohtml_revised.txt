#!/usr/bin/env python3
"""
Convert 2019-2020 FOMC PDF transcripts to HTML
"""

import os
import glob
import pdftotree

PDF_DIR = 'data/pdfs'
HTML_DIR = 'fomc_transcript/htmls'

os.makedirs(HTML_DIR, exist_ok=True)

# Get 2019-2020 PDFs only
pdf_files = glob.glob(f'{PDF_DIR}/FOMC201*.pdf') + glob.glob(f'{PDF_DIR}/FOMC202*.pdf')
print(f"Found {len(pdf_files)} PDFs to convert")

for pdf_file in pdf_files:
    # Extract date: FOMC20190130meeting.pdf -> 20190130
    filename = os.path.basename(pdf_file)
    date_str = filename[4:12]
    
    html_output = f"{HTML_DIR}/{date_str}.html"
    
    if os.path.exists(html_output):
        print(f"  ✓ Already exists: {date_str}")
        continue
    
    print(f"  Converting: {date_str}...", end=" ")
    try:
        pdftotree.parse(pdf_file, html_path=html_output)
        print("✓")
    except Exception as e:
        print(f"✗ Error: {e}")

print("\n✅ Done converting PDFs to HTML!")
print(f"Now run your HTML parser script to create CSVs.")
